createElement = function createElement(html) {
  var root = document.createElement('div');
 
  root.innerHTML = html
    .replace(/^\s+|\s+$/gm, '')
    .replace(/\s*\n\s*</gm, '<')
    .replace(/>\s*\n\s*/gm, '>')
    .trim();
 
  return root.firstChild;
};
